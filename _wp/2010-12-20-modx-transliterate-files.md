    	function getDirSize($dir) {
    		$dirSize=0;
    		if ($dh=@opendir($dir)) {
    			while ($file=@readdir($dh)) {
    				if (($file!=".")&&($file!="..")) {
    					if (is_dir($dir."/".$file)) {
    						$tmp_dirSize=$this->getDirSize($dir."/".$file);
    						if ($tmp_dirSize!==false) $dirSize+=$tmp_dirSize;
    					} else {
    						$dirSize+=filesize($dir."/".$file);
    					}
    				}
    			}
    			@closedir($dh);
    		} else {
    			return false;
    		}
    
    		return $dirSize;
